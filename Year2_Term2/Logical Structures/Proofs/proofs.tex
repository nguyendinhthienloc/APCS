\subsection*{(a) $\forall x\, P(x) \vdash \forall y\, P(y)$}
% Strategy: open a fresh-variable box for y_0, derive P(y_0) via ∀e,
% then close and apply ∀i.
% ─────────────────────────────────────────────────────────────────────────────

  \forall x\, P(x) & premise \\
  \begin{subproof}
    y_0 & fresh variable \\
    P(y_0) & $\forall$e, 1
  \end{subproof}
  \forall y\, P(y) & $\forall$i, 2--3
\end{logicproof}

\subsection*{(b) $\forall x\,(P(x) \to Q(x)) \vdash
             (\forall x\,\lnot Q(x)) \to (\forall x\,\lnot P(x))$}
% Strategy:
%   Outer box  → assume ∀x ¬Q(x)  (for →i)
%   Middle box → fresh x_0         (for ∀i)
%   Inner box  → assume P(x_0)     (for ¬i, reach ⊥)
% ─────────────────────────────────────────────────────────────────────────────

\begin{logicproof}{3}
  \forall x\,(P(x) \to Q(x)) & premise \\
  \begin{subproof}
    \forall x\,\lnot Q(x) & assumption \\
    \begin{subproof}
      x_0 & fresh variable \\
      P(x_0) \to Q(x_0) & $\forall$e, 1 \\
      \lnot Q(x_0) & $\forall$e, 2 \\
      \begin{subproof}
        P(x_0) & assumption \\
        Q(x_0) & $\to$e, 4, 6 \\
        \bot & $\lnot$e, 5, 7
      \end{subproof}
      \lnot P(x_0) & $\lnot$i, 6--8
    \end{subproof}
    \forall x\,\lnot P(x) & $\forall$i, 3--9
  \end{subproof}
  (\forall x\,\lnot Q(x)) \to (\forall x\,\lnot P(x)) & $\to$i, 2--10
\end{logicproof}

\subsection*{(c) $\forall x\,(P(x) \to \lnot Q(x)) \vdash
             \lnot(\exists x\,(P(x) \land Q(x)))$}
% Strategy:
%   Outer box → assume ∃x(P(x)∧Q(x))          (for ¬i)
%   Inner box → witness x_0 with P(x_0)∧Q(x_0) (for ∃e)
%               derive ⊥ inside, then push ⊥ out via ∃e, then ¬i
% ─────────────────────────────────────────────────────────────────────────────

\begin{logicproof}{2}
  \forall x\,(P(x) \to \lnot Q(x)) & premise \\
  \begin{subproof}
    \exists x\,(P(x) \land Q(x)) & assumption \\
    x_0 &  \\
    P(x_0) \land Q(x_0) & $\exists$e 2,3 \\
    P(x_0) \to \lnot Q(x_0) & $\forall$e, 1 \\
    P(x_0) & $\land$e$_1$, 4 \\
    Q(x_0) & $\land$e$_2$, 4 \\
    \lnot Q(x_0) & $\to$e, 5, 6 \\
    \bot & $\lnot$e, 7,8
  \end{subproof}
  \lnot(\exists x\,(P(x) \land Q(x))) & $\lnot$i, 2--9
\end{logicproof}
