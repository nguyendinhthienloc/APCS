        \begin{subproof}

            \forall y (P(x_0) \to Q(y)) & $\forall E$ (2) \\
            P(x_0) \to Q(y_0) & $\forall E$ (5) \\
            Q(y_0) & $\to E$ (3, 6)
        \end{subproof}
        \forall y Q(y) & $\forall I$ (4--7)
    \end{subproof}
    \forall y Q(y) & $\exists E$ (1, 3--8)
